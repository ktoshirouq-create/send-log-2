function doGet(e) { try { var ss = SpreadsheetApp.getActiveSpreadsheet(); var climbsSheet = ss.getSheetByName("Climbs"); var sessionsSheet = ss.getSheetByName("Sessions"); var climbs = getSheetDataAsObjects(climbsSheet); var sessions = getSheetDataAsObjects(sessionsSheet); return ContentService.createTextOutput(JSON.stringify({ status: 'success', climbs: climbs, sessions: sessions })).setMimeType(ContentService.MimeType.JSON); } catch (error) { return ContentService.createTextOutput(JSON.stringify({ status: 'error', message: error.toString() })).setMimeType(ContentService.MimeType.JSON); } } function doPost(e) { try { var payload = JSON.parse(e.postData.contents); var ss = SpreadsheetApp.getActiveSpreadsheet(); // Auto-create sheets if they somehow got deleted var climbsSheet = ss.getSheetByName("Climbs") || ss.insertSheet("Climbs"); var sessionsSheet = ss.getSheetByName("Sessions") || ss.insertSheet("Sessions"); if (payload.action === "sync_all") { // 1. Process Deletions First if (payload.deletedClimbs && payload.deletedClimbs.length > 0) { deleteRows(climbsSheet, "ClimbID", payload.deletedClimbs); } if (payload.deletedSessions && payload.deletedSessions.length > 0) { deleteRows(sessionsSheet, "SessionID", payload.deletedSessions); } // 2. Process Session Updates/Inserts if (payload.sessions && payload.sessions.length > 0) { upsertData(sessionsSheet, payload.sessions, "SessionID"); } // 3. Process Climb Updates/Inserts if (payload.climbs && payload.climbs.length > 0) { upsertData(climbsSheet, payload.climbs, "ClimbID"); } return ContentService.createTextOutput(JSON.stringify({status: 'success'})).setMimeType(ContentService.MimeType.JSON); } } catch (error) { return ContentService.createTextOutput(JSON.stringify({ status: 'error', message: error.toString() })).setMimeType(ContentService.MimeType.JSON); } } // --- HELPER FUNCTIONS --- function getSheetDataAsObjects(sheet) { if (!sheet) return []; var data = sheet.getDataRange().getValues(); if (data.length < 2) return []; var headers = data[0]; var result = []; for (var i = 1; i < data.length; i++) { var obj = {}; for (var j = 0; j < headers.length; j++) { if (data[i][j] !== "") { // Ignore empty cells to save payload size obj[headers[j]] = data[i][j]; } } result.push(obj); } return result; } function upsertData(sheet, items, idKey) { var data = sheet.getDataRange().getValues(); var headers = data.length > 0 ? data[0] : []; // 1. Scan incoming items and auto-create any missing columns var newHeaders = []; items.forEach(function(item) { Object.keys(item).forEach(function(key) { // Ignore the local sync flag, only save real data if (key !== '_synced' && headers.indexOf(key) === -1 && newHeaders.indexOf(key) === -1) { newHeaders.push(key); } }); }); if (newHeaders.length > 0) { if (headers.length === 0) { headers = newHeaders; sheet.appendRow(headers); } else { sheet.getRange(1, headers.length + 1, 1, newHeaders.length).setValues([newHeaders]); headers = headers.concat(newHeaders); } } // 2. Map existing IDs to their Row Numbers to enable fast Updates var idMap = {}; var idColIndex = headers.indexOf(idKey); if (idColIndex !== -1 && data.length > 1) { for (var i = 1; i < data.length; i++) { idMap[String(data[i][idColIndex])] = i + 1; // +1 because sheet rows start at 1 } } // 3. Insert or Update the rows items.forEach(function(item) { var rowData = headers.map(function(h) { return item[h] !== undefined ? item[h] : ""; }); var id = String(item[idKey]); if (idMap[id]) { // UPDATE: ID exists, overwrite the row sheet.getRange(idMap[id], 1, 1, rowData.length).setValues([rowData]); } else { // INSERT: ID doesn't exist, add it to the bottom sheet.appendRow(rowData); idMap[id] = sheet.getLastRow(); // Add to map in case of duplicates in payload } }); } function deleteRows(sheet, idKey, idsToDelete) { var data = sheet.getDataRange().getValues(); if (data.length < 2) return; var headers = data[0]; var idColIndex = headers.indexOf(idKey); if (idColIndex === -1) return; var rowsToDelete = []; for (var i = 1; i < data.length; i++) { if (idsToDelete.indexOf(String(data[i][idColIndex])) !== -1) { rowsToDelete.push(i + 1); } } // Delete from bottom up so row indexes don't shift during deletion rowsToDelete.sort(function(a, b) { return b - a; }).forEach(function(rowIndex) { sheet.deleteRow(rowIndex); }); }
